Node.js Example

const axios = require('axios'); async function generateClips() { try { const response = await axios.post('http://localhost:3001/api/generate', { videoUrl: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ', customDuration: 30, plan: 'business', portraitMode: true }); console.log('Generated clips:', response.data.clips); } catch (error) { console.error('Error:', error.response?.data || error.message); } } generateClips();

Python Example

import requests import json def generate_clips(): url = "http://localhost:3001/api/generate" payload = { "videoUrl": "https://www.youtube.com/watch?v=dQw4w9WgXcQ", "customDuration": 30, "plan": "business", "portraitMode": True } try: response = requests.post(url, json=payload) response.raise_for_status() data = response.json() print("Generated clips:", json.dumps(data["clips"], indent=2)) except requests.exceptions.RequestException as e: print(f"Error: {e}") generate_clips()
